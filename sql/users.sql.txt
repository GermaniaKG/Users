CREATE TABLE `users` (
	`id` int(11) unsigned NOT NULL AUTO_INCREMENT,
	`user_name` varchar(64) NOT NULL DEFAULT '',
	`user_last_name` varchar(256) DEFAULT NULL,
	`user_first_name` varchar(256) DEFAULT NULL,
	`user_email` varchar(256) DEFAULT NULL,
	`user_login_name` varchar(256) DEFAULT NULL,
	`password` varchar(128) CHARACTER SET utf8 COLLATE utf8_bin NOT NULL DEFAULT '',
	`user_display_name` varchar(64) NOT NULL,
	`api_key` varchar(256) DEFAULT NULL,
	`created` datetime NOT NULL,
	`updated` timestamp NOT NULL DEFAULT '0000-00-00 00:00:00' ON UPDATE CURRENT_TIMESTAMP,
	`is_active` int(2) DEFAULT '-1',
	PRIMARY KEY (`id`),
	UNIQUE KEY `unique_email` (`user_email`),
	UNIQUE KEY `unique_login_name` (`user_login_name`),
	UNIQUE KEY `unqiue_api_key` (`api_key`),
	KEY `index_username` (`user_name`),
	KEY `index_email` (`user_email`),
	KEY `index_api_key` (`api_key`)
) ENGINE=MyISAM DEFAULT CHARSET=utf8;
